function updateUI() {
  document.getElementById("env").innerText = Math.max(0, Math.min(100, env));
  document.getElementById("money").innerText = Math.max(0, Math.min(100, money));
  document.getElementById("happy").innerText = Math.max(0, Math.min(100, happy));
  document.getElementById("reputation").innerText = Math.max(0, Math.min(100, reputation));
  
  const progress = ((step / events.length) * 100);
  document.getElementById("progress").style.width = progress + "%";
}

function loadEvent() {
  if (step >= events.length) {
    endGame();
    return;
  }
  
  const event = events[step];
  document.getElementById("question").innerText = event.q;
  
  const optionsContainer = document.getElementById("options");
  optionsContainer.innerHTML = "";
  
  event.options.forEach((option, index) => {
    const btn = document.createElement("button");
    btn.innerText = `${index + 1}. ${option.text}`;
    btn.onclick = () => choose(index);
    optionsContainer.appendChild(btn);
  });
}

function choose(optionIndex) {
  const event = events[step];
  const effect = event.options[optionIndex].effect;
  
  env += effect[0];
  money += effect[1];
  happy += effect[2];
  reputation += effect[3];
  
  // Cek game over dini
  if (money <= 0 || happy <= 20 || env <= 20) {
    endGame(true);
    return;
  }
  
  step++;
  updateUI();
  loadEvent();
}

function endGame(earlyEnd = false) {
  document.getElementById("gameCard").style.display = "none";
  const endDiv = document.getElementById("end");
  endDiv.style.display = "block";
  
  let title = "", desc = "", icon = "";
  
  if (earlyEnd) {
    title = "💥 KESEPAKATAN GAGAL!";
    desc = "Keputusanmu menyebabkan kegagalan dini. Coba strategi yang lebih seimbang!";
    icon = "💥";
  } else if (env >= 85 && happy >= 85 && reputation >= 85 && money >= 40) {
    title = "🏆 SEKOLAH HIJAU EKSELEN!";
    desc = "Sekolahmu menjadi contoh nasional! Mendapat penghargaan internasional dan dana hibah besar. 🌍✨";
    icon = "🏆";
  } else if (env >= 75 && happy >= 75 && reputation >= 75) {
    title = "🌟 SEKOLAH HIJAU SUKSES!";
    desc = "Pencapaian luar biasa! Sekolah ramah lingkungan dengan siswa bahagia dan reputasi tinggi! 🎉";
    icon = "🌟";
  } else if (env >= 70 && happy >= 70 && reputation >= 70) {
    title = "✅ SEKOLAH HIJAU LULUS!";
    desc = "Target tercapai! Sekolahmu sudah ramah lingkungan dan sustainable. Bagus sekali! 👍";
    icon = "✅";
  } else if (money >= 80) {
    title = "💰 SEKOLAH KAYA!";
    desc = "Dana melimpah tapi lingkungan kurang terjaga. Fokuskan lebih pada sustainability!";
    icon = "💰";
  } else {
    title = "⚠️ PERLU PERBAIKAN";
    desc = "Ada kemajuan tapi belum optimal. Evaluasi kembali strategi lingkunganmu!";
    icon = "⚠️";
  }
  
  endDiv.innerHTML = `
    <div class="ending-icon">${icon}</div>
    <div class="end-title">${title}</div>
    <div class="end-desc">${desc}</div>
    <div style="margin: 20px 0; font-size: 16px;">
      Final Score:<br>
      Lingkungan: ${env} | Dana: ${money} | Siswa: ${happy} | Reputasi: ${reputation}
    </div>
    <button class="restart-btn" onclick="restartGame()">🔄 Main Lagi</button>
  `;
}

function restartGame() {

  document.getElementById("end").style.display = "none";
